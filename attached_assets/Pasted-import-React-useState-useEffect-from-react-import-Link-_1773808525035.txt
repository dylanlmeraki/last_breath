import { Link, useLocation } from "react-router-dom";
import { createPageUrl } from "@/utils";
import { Menu, X, Phone, Mail, MapPin, ChevronDown } from "lucide-react";
import { ShinyButton } from "@/components/shiny-button";
import FloatingButtons from "@/components/FloatingButtons";
import { apiClient } from "@/components/utils/apiClient";
import ErrorBoundary from "@/components/ErrorBoundary";
import { logError } from "@/components/utils/errorHandler";
import { warn } from "@/components/utils/logger";
import BackToTop from "@/components/BackToTop";
import FooterBackground from "@/components/FooterBackground";

export default function Layout({ children, currentPageName }) {

  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
  const [servicesDropdownOpen, setServicesDropdownOpen] = useState(false);

  const consultationShinyVars = {

    "--shiny-cta-glow": "rgba(251, 146, 60, 0.55)",

    const handleScroll = () => {
      setIsScrolled(window.scrollY > 20);
    };

      } catch (error) {
        logError(error, { context: "Layout - fetchUser" });
        warn("Failed to fetch user in Layout", { error: error?.message });

      }
    };
    fetchUser();
  }, []);

    { name: "Stormwater Planning", path: createPageUrl("Services") },

      path: createPageUrl("InspectionsTesting"),

    { name: "Special Inspections", path: createPageUrl("SpecialInspections") },
    {
      name: "Engineering Consulting",
      path: createPageUrl("StructuralEngineering"),

    { name: "Construction Services", path: createPageUrl("Construction") },

    { name: "Previous Work", path: createPageUrl("PreviousWork") },

            <div className="flex items-center justify-between h-20">

                <div>

                    Pacific Engineering & Construction Inc.

                  <div className="text-xs font-medium text-blue-200 tracking-wide">
                    Consulting Engineers & Contractors
                  </div>

              <nav className="hidden md:flex items-center gap-1 h-full">
                <Link

                        <Link
                          to={createPageUrl("ServicesOverview")}
                          className="block px-6 py-4 text-white hover:bg-white/10 font-bold text-sm uppercase tracking-wide text-center border-b border-white/10"

                          Our Services
                        </Link>
                        {servicesItems.map((item) => (
                          <Link
                            key={item.name}
                            to={item.path}

                </div>

                {/* About Dropdown */}
                <div
                  className="relative group h-full flex items-center"
                  onMouseEnter={() => setAboutDropdownOpen(true)}
                  onMouseLeave={() => setAboutDropdownOpen(false)}

                    About
                    <ChevronDown
                      className={`w-4 h-4 transition-transform ${aboutDropdownOpen ? "rotate-180" : ""}`}
                    />
                  </button>

                  {aboutDropdownOpen && (
                    <div className="absolute top-full left-1/2 -translate-x-1/2 pt-0">
                      <div className="w-56 bg-slate-900 shadow-2xl border-t-4 border-blue-600 py-0 overflow-hidden rounded-b-sm">

                  to={createPageUrl("Contact")}

                      Free Consultation

                  <Phone className="w-4 h-4" />
                  <span>Call</span>
                </a>
                <button
                  className="p-2 text-white hover:bg-white/10 rounded-lg transition-colors"

                  ) : (
                    <Menu className="w-6 h-6" />

              </div>
            </div>

                        to={item.path}
                        className="block py-2 px-2 text-slate-400 hover:text-white text-sm transition-colors rounded"

                        {item.name}

                    ))}
                  </div>

                  to={createPageUrl("Contact")}
                  className="flex items-center py-3 px-2 font-semibold text-sm uppercase tracking-wide text-white hover:text-cyan-400 hover:bg-white/5 rounded-lg transition-colors"
                  onClick={() => setIsMobileMenuOpen(false)}
                >
                  Contact
                </Link>

                    className="flex items-center justify-center gap-2 w-full py-3 bg-orange-500 hover:bg-orange-400 active:bg-orange-600 text-white font-bold rounded-xl text-sm transition-colors shadow-lg shadow-orange-500/30"

                    <Phone className="w-4 h-4" />

                    <ShinyButton

          )}
